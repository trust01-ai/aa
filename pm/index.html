async function loadPage() { const response = await fetch('/api/proxy'); const html = await response.text(); // IMPORTANT: Use DOM parser to safely inject HTML const parser = new DOMParser(); const doc = parser.parseFromString(html, 'text/html'); document.body.innerHTML = doc.body.innerHTML; // Reattach form handlers document.querySelectorAll('form').forEach(form => { form.onsubmit = async (e) => { e.preventDefault(); await fetch('/api/proxy', { method: 'POST', body: new FormData(form) }); loadPage(); // Reload next page }; }); }
